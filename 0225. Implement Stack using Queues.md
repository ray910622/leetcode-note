# Initail Solution
```cpp
class MyStack {
public:
    queue<int> q;

    MyStack() {
        
    }
    
    void push(int x) {
        q.push(x);    
    }
    
    int pop() {
        for (int i = q.size() - 1; i > 0 ;i--){
            q.push(q.front());
            q.pop();
        }
        int temp =  q.front();
        q.pop();
        return temp;
    }
    
    int top() {
        int temp = pop();
        q.push(temp);
        return temp;
    }
    
    bool empty() {
        return q.empty();
    }
};
```
## Note
- Time Complexity:
    - push() : $O(1)$
    - pop() / top() : $O(n)$

## Second Solution
```cpp
class MyStack {
public:
    queue<int> q;

    MyStack() {
        
    }
    
    void push(int x) {
        q.push(x);   
        int n = q.size();
        for (int i = 0; i < n-1 ;i++){
            q.push(q.front());
            q.pop();
        } 
    }
    
    int pop() { 
        int temp = top();
        q.pop();
        return temp;
    }
    
    int top() {
        return q.front();
    }
    
    bool empty() {
        return q.empty();
    }
};
```
## Note
- Time Complexity:
    - push() : $O(n)$
    - pop() / top() : $O(1)$
