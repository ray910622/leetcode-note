```cpp
class Solution {
public:
    bool isValid(string s) {
        stack<char> st;
        for (char ch : s){
            if (ch == '(' || ch == '{' || ch == '['){       //如果是左括號就push進st
                st.push(ch);
            }else{                                          //如果為空卻又不是左括號，代表有錯
                if ( st.empty() || (ch == ')' && st.top() != '(') ||    //如果是其他值，對應看看是不是符合最後放入的
                (ch == '}' && st.top() != '{') || (ch == ']' && st.top() != '[') ){
                    return false;
                }
                st.pop();   //有成功對應
            }
        }
        return st.empty();  //如果迴圈跑完為空，代表全部都有對應，正確；否則有剩，回傳錯誤
    }
};
```
