class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<int> res;
        unordered_map<int, int> mmap;
        for (int i = 0; i < nums.size(); i++) {
            auto it = mmap.find(target - nums[i]);
            if (it == mmap.end()) {
                mmap.insert({nums[i], i});
            } else {
                res.push_back(i);
                res.push_back(it->second);
                return res;
            }
        }
        return res;
    }
};
